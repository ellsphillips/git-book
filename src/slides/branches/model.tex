\newcommand\commit[2][master]{
	\node[commit, #1] (#2) {};
	\node[clabel] at (#2) {};
}
\newcommand\ghost[1]{
	\coordinate (#1);
}
\newcommand\connect[2]{
	\draw (#1) to[out=-90,in=90] (#2);
}
